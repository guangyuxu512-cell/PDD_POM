@staticmethod
def _提取金额(值: Any) -> float:
if isinstance(值, (int, float)):
return float(值)
匹配 = re.search(r"(d+(?:.d+)?)", str(值 or ""))
return float(匹配.group(1)) if 匹配 else 0.0
def _找按钮(self, 可用按钮: list[str], 优先级列表: list[str]) -> str:
for 关键词 in 优先级列表:
for 按钮 in 可用按钮:
if 关键词 in 按钮:
return 按钮
return ""

self,
详情: dict,
可用按钮: list[str],
售后配置: dict,
队列记录: dict,
结果: dict,
) -> dict:
"""退货退款决策，按可用按钮分支处理。"""
── 分支1: 有「同意拒收退款」按钮 → 刚申请 / 未发货 ──
拒收按钮 = self._找按钮(可用按钮, ["同意拒收退款", "同意拒收后退款"])
if 拒收按钮:
return self._处理_拒收退款分支(详情, 拒收按钮, 售后配置, 结果)
── 分支2: 有「同意退货」按钮 → 同意退货 ──
同意退货按钮 = self._找按钮(可用按钮, ["同意退货"])
if 同意退货按钮:
结果["操作"] = "同意退货"
结果["目标按钮"] = 同意退货按钮
return 结果
── 分支3: 有「同意退款」按钮 → 检查退货物流 ──
同意退款按钮 = self._找按钮(可用按钮, ["同意退款"])
if 同意退款按钮:
return self._处理_退货物流分支(详情, 可用按钮, 售后配置, 队列记录, 结果)
── 兜底: 没有可操作按钮 ──
结果["操作"] = "跳过"
结果["人工原因"] = "退货退款订单无可操作按钮"
return 结果

结果["需要微信通知"] = True
结果["需要飞书通知"] = True
if "签收" in 物流状态:

结果["操作"] = "人工处理"

未签收 → 同意拒收退款 + 通知拦截

结果["飞书通知内容"] = 结果["微信通知内容"]

def _处理_退货物流分支(
self,
详情: dict,
可用按钮: list[str],
售后配置: dict,
队列记录: dict,
结果: dict,
) -> dict:
"""有「同意退款」按钮时，按退货物流状态决定后续动作。"""
del 队列记录  # 当前未用，保留参数便于后续扩展
退货物流 = dict(详情.get("退货物流信息") or {})
有退货物流 = bool(退货物流.get("有退货物流"))
if not 有退货物流:
结果["操作"] = "人工处理"
结果["人工原因"] = "无退货物流单号，需人工确认"
return 结果
白名单 = list(售后配置.get("退货物流白名单") or [])
等待时间 = dict(
售后配置.get("退货等待时间")
or {"刚发出": 3, "中途运输": 1, "到达目的市": 0.25}
)
需要入库校验 = bool(售后配置.get("需要入库校验", False))
自动退款上限 = float(售后配置.get("自动退款金额上限", 50) or 50)
所有地区词: list[str] = []
for 规则 in 白名单:
所有地区词.extend(list(规则.get("地区关键词") or []))
轨迹全文 = str(退货物流.get("轨迹全文", ""))
物流阶段 = self._判断物流阶段(轨迹全文, 所有地区词)
等待阶段
if 物流阶段 in ("刚发出", "中途运输", "到达目的市"):
结果["操作"] = "等待"
结果["下次处理天数"] = 等待时间.get(物流阶段, 1)
return 结果
派送中 / 已签收
if 物流阶段 in ("派送中", "已签收"):
匹配 = self._匹配白名单(退货物流, 白名单)
if not 匹配["派件人匹配"]:
结果["操作"] = "人工处理"
结果["人工原因"] = "派件人不在白名单"
return 结果
if 需要入库校验:
订单号 = 详情.get("订单编号") or 详情.get("订单号") or ""
结果["操作"] = "等待验货"
结果["需要备注"] = True
结果["备注内容"] = "退回件已到，待入库校验"
结果["需要飞书通知"] = True
结果["飞书通知内容"] = f"订单{订单号} 退回件到达，请入库校验后在前端确认"
return 结果
金额 = self._提取金额(详情.get("退款金额", 0))
if 金额 <= 自动退款上限:
结果["操作"] = "同意退款"
结果["目标按钮"] = self._找按钮(可用按钮, ["同意退款"])
return 结果
订单号 = 详情.get("订单编号") or 详情.get("订单号") or ""
结果["操作"] = "等待验货"
结果["需要备注"] = True
结果["备注内容"] = f"退回件已到，金额{金额}元超限{自动退款上限}，待验货"
结果["需要飞书通知"] = True
结果["飞书通知内容"] = f"订单{订单号} 金额{金额}超限，请验货后确认"
return 结果
未知物流阶段
结果["操作"] = "人工处理"
结果["人工原因"] = f"物流状态未知: {轨迹全文[:80]}"
return 结果

@staticmethod
def _空结果(售后配置: dict) -> dict:
"""生成空的决策结果模板。"""
return {
"操作": "人工处理",
"目标按钮": "",
"备选按钮": [],
"弹窗偏好": dict(售后配置.get("弹窗偏好") or {}),
"需要备注": False,
"备注内容": "",
"需要飞书通知": False,
"飞书通知内容": "",
"人工原因": "",
"下次处理天数": None,
"需要微信通知": False,
"微信通知内容": "",
"需要飞书写入": False,
}
@staticmethod
def _判断物流阶段(轨迹全文: str, 目的市关键词: list[str]) -> str:
if not 轨迹全文:
return "未知"
if "签收" in 轨迹全文 or "已签收" in 轨迹全文:
return "已签收"
if "派件" in 轨迹全文 or "派送" in 轨迹全文 or "正在为您" in 轨迹全文:
return "派送中"
for 关键词 in 目的市关键词:
if (
f"到达{关键词}" in 轨迹全文
or f"到{关键词}" in 轨迹全文
or f"{关键词}市" in 轨迹全文
):
return "到达目的市"
if "运输" in 轨迹全文 or "转运" in 轨迹全文 or "发往" in 轨迹全文:
return "中途运输"
if "揽收" in 轨迹全文 or "已发出" in 轨迹全文 or "已揽件" in 轨迹全文:
return "刚发出"
return "未知"
@staticmethod
def _匹配白名单(退货物流: dict, 白名单: list[dict]) -> dict:
"""匹配退货物流是否命中白名单。"""
快递公司 = str(退货物流.get("退货快递公司", ""))
轨迹全文 = str(退货物流.get("轨迹全文", ""))
派件人 = str(退货物流.get("派件人", ""))
网点 = str(退货物流.get("网点", ""))
合并文本 = f"{轨迹全文} {派件人} {网点}"
for 规则 in 白名单:
规则快递 = str(规则.get("快递公司", "*"))
if 规则快递 != "*" and 规则快递 not in 快递公司:
continue
地区匹配 = any(关键词 in 合并文本 for 关键词 in list(规则.get("地区关键词") or []))
派件人匹配 = any(名字 in 合并文本 for 名字 in list(规则.get("派件人") or []))
if 地区匹配 or 派件人匹配:
return {
"地区匹配": 地区匹配,
"派件人匹配": 派件人匹配,
"匹配的规则": 规则,
}
return {"地区匹配": False, "派件人匹配": False, "匹配的规则": None}

### 5) 验收标准

- **`决策()` 入口只有两条路径**：退货退款 → `_决策_退货退款()`，其他 → 人工处理
- **`_决策_拒收退款()` 不再存在**，其逻辑在 `_处理_拒收退款分支()` 里，被 `_决策_退货退款()` 内部调用
- **`_决策_退款()` 保留但标注 Phase2**，`决策()` 不会调用它
- **未签收 + 拒收退款启用 → 操作变为 "同意拒收退款"**（之前是"人工处理"，这是修复）
- 单元测试用例：
​
测试1: 退款类型 → 人工处理
assert (await 引擎.决策({"售后类型": "退款"}, 配置, {}))["操作"] == "人工处理"
测试2: 退货退款 + 同意拒收退款按钮 + 未签收 → 同意拒收退款
assert (await 引擎.决策({"售后类型": "退货退款", "可用按钮列表": ["同意拒收退款"], "物流最新状态": "运输中"}, 配置, {}))["操作"] == "同意拒收退款"
测试3: 补寄 → 人工处理
assert (await 引擎.决策({"售后类型": "补寄"}, 配置, {}))["操作"] == "人工处理"

### 6) 风险与注意事项

| 风险 | 应对 |
|---|---|
| `_决策_退款()` 里有被其他文件 import 的可能 | 搜索确认无外部引用，安全保留为 Phase2 |
| 原来 `决策()` 对纯退款类型也会走拒收退款判断 | 现在统一由 `_决策_退货退款()` 内部根据按钮分支，不再单独判断 |
| `_空结果()` 提取为独立方法 | 避免每个分支重复构造结果模板，更干净 |
